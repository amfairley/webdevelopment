outerLoop:
for (let x = 0; x < 5; x++) {
    for (let y = 0; y < 5; y++) {
        if (y === 3) {
            console.log("Breaking outer loop from inner loop");
            break outerLoop; // Exits the outer loop
        }
        console.log(`x: ${x}, y: ${y}`);
    }
}
