// Getter Syntax
get propertyName() {
    return this._propertyName; // _propertyName is the actual property name
}

// Getter Example
const person = {
    _name: 'Adam',
    get name() {
    return this._name; // Returns the value of the private _name property
    }
};

console.log(person.name); // Outputs: 'Adam'
