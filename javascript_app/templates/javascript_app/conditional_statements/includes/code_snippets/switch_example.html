let result = 75;

    console.log("A");

    Console.log ("B");

    Console.log("C");

// After
let result = 75;

switch (true) {
    case (result > 79):
        console.log("A");
        break;
    case (result > 69):
        console.log("B");
        break;
    default:
        console.log("C");
}
