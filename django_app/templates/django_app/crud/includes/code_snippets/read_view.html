from django.shortcuts import render
from .models import Item

def get_todo_list(request):
	# Fetch all the items
    items = Item.objects.all()  # Fetch all items
	# Add to context
	context = {
		'items': items,
}
    return render(request, 'todo/todo_list.html', context)
