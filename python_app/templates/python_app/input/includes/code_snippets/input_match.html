choice = input("Do you want to start, stop, or quit? ").lower()

match choice:
    case "start" | "go":
        print("Starting the program...")
    case "stop":
        print("Stopping the program.")
    case "quit":
        print("Exiting. Goodbye!")
    case _:
        print("Invalid option. Please try again.")
