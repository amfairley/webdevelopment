class Dog:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    # Setter method to update the age
    def set_age(self, new_age):
        self.age = new_age

    # Getter method to return value
    def get_name(self):
        return self.name

    def __str__(self):
        return f"{self.name}, age {self.age}"
